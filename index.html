            - "MESSAGE={{ '{{ message }}' }}"

ansible/main.yml
- hosts: foo
  gather_facts: false
  tasks:
    - name: Install Packages
      yum:
        name: "{{ item }}"
      with_items:
        - vim
        - wget
    - template:
        src: "templates/entrypoint.j2"
        dest: "/usr/bin/entrypoint"
        group: "root"
        mode: "ug+x"
entrypoint.js
#!/bin/sh
USER_ID=$(id -u)

echo "${MESSAGE}"

exec "$@"
